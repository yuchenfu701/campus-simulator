# 🔐 账号登录与自动保存功能说明

## 📋 功能概述

### 核心功能
1. ✅ **完整的账号登录系统**
   - 基于手机号的唯一账号识别
   - 新账号自动初始化
   - 多账号数据完全隔离

2. ✅ **自动保存机制**
   - 退出时自动保存
   - 定期自动保存
   - 页面关闭前自动保存
   - 保存确认提示

function initUserPointsData(userId, nickname) {
    let pointsData = JSON.parse(localStorage.getItem('campus_points_system') || '{}');
    
    if (!pointsData.students) {
        pointsData.students = {};
    }
    
    // 检查是否为新用户
    if (!pointsData.students[userId]) {
        // 创建初始数据
        pointsData.students[userId] = {
            id: userId,
            name: nickname,
            points: 100,           // 初始积分
            totalEarned: 100,
            inventory: [],         // 空背包

                hair: 'none',
                glasses: 'none',
                hat: 'none'
            },
            achievements: [],
            tasks: {
                daily: {},
                weekly: {}
            }
        };
        
        localStorage.setItem('campus_points_system', JSON.stringify(pointsData));
        console.log(`✅ 新用户 ${nickname} 数据已初始化`);
    }
}

### 2. 自动保存系统

#### 主菜单自动保存 (`main-menu.html`)

```javascript
// 自动保存函数
function autoSave() {
    try {
        const userData = JSON.parse(localStorage.getItem('campus_user') || '{}');
        if (userData.userId) {
            const pointsSystem = new PointsSystem();
            pointsSystem.setCurrentUser(userData.userId);
            pointsSystem.savePointsData();
            console.log('💾 自动保存完成');
        }
    } catch (e) {
        console.error('❌ 自动保存失败:', e);
    }
}

// 页面关闭前保存
window.addEventListener('beforeunload', () => {
    autoSave();
});

// 定期自动保存（每30秒）
setInterval(autoSave, 30000);
```

#### 退出登录保存 (`main-menu.html`)

```javascript
function logout() {
    if (confirm('确定要退出登录吗？所有数据将自动保存。')) {
        // 1. 确保积分系统数据已保存
        const pointsSystem = new PointsSystem();
        if (pointsSystem.currentUser) {
            pointsSystem.savePointsData();
            console.log('✅ 积分数据已保存');
        }
        
        // 2. 保存退出时间
        const userData = JSON.parse(localStorage.getItem('campus_user') || '{}');
        if (userData.userId) {
            userData.lastLogoutTime = new Date().toISOString();
            localStorage.setItem('campus_user', JSON.stringify(userData));
        }
        
        // 3. 显示保存成功提示
        showSaveMessage('✅ 数据已保存，正在退出...');
        
        // 4. 延迟跳转
        setTimeout(() => {
            localStorage.removeItem('campus_user');
            window.location.href = 'login.html';
        }, 800);
    }
}
```

#### 游戏内自动保存 (`index.html`)

##### 自动保存函数
```javascript
function autoSaveGameData() {
    try {
        // 1. 保存游戏进度
        if (window.simulator && window.simulator.player) {
            const gameState = {
                position: window.simulator.player.position,
                currentArea: window.simulator.currentArea,
                inventory: window.simulator.player.inventory,
                timestamp: new Date().toISOString()
            };
            localStorage.setItem('campus_game_state', JSON.stringify(gameState));
        }
        
        // 2. 保存积分系统数据
        const userData = JSON.parse(localStorage.getItem('campus_user') || '{}');
        if (userData.userId) {
            const pointsSystem = new PointsSystem();
            pointsSystem.setCurrentUser(userData.userId);
            pointsSystem.savePointsData();
        }
        
        console.log('💾 游戏数据自动保存成功');
        return true;
    } catch (e) {
        console.error('❌ 自动保存失败:', e);
        return false;
    }
}
```

### 3. 保存提示系统

#### 显示保存消息
```javascript
function showSaveMessage(message) {
    const saveMsg = document.createElement('div');
    saveMsg.style.cssText = `
        position: fixed;
        top: 50%;
        left: 50%;
        transform: translate(-50%, -50%);
        background: linear-gradient(135deg, #667eea, #764ba2);
        color: white;
        padding: 30px 50px;
        border-radius: 16px;
        font-size: 18px;
        z-index: 100000;
        box-shadow: 0 8px 32px rgba(0,0,0,0.3);
    `;
    saveMsg.textContent = message;
    document.body.appendChild(saveMsg);
    
    // 1.5秒后自动移除
    setTimeout(() => {
        if (saveMsg.parentNode) {
            saveMsg.remove();
        }
    }, 1500);
}
```
