### 3. **添加goToPage函数**

**添加的函数**：
```javascript
function goToPage(page) {
    console.log('🔄 尝试跳转到页面:', page);
    
    const pages = {
        'mall': 'points-mall.html',
        'inventory': 'points-inventory.html',
        'avatar': 'points-avatar.html',
        'rank': 'points-rank.html',
        'tasks': 'points-tasks.html'
    };
    
    const titles = {
        'mall': '💎 积分商城',
        'inventory': '🎒 我的背包',
        'avatar': '👤 虚拟形象',
        'rank': '🏆 排行榜',
        'tasks': '📋 任务中心'
    };
    
    // 检查页面是否存在
    if (!pages[page]) {
        console.error('❌ 未知的页面:', page);
        alert('❌ 页面不存在');
        return;
    }
    
    // 如果当前页面就是目标页面，不跳转
    const currentPage = window.location.pathname.split('/').pop();
    if (pages[page] === currentPage) {
        console.log('ℹ️ 已在目标页面');
        return;
    }
    
    // 如果在iframe中，尝试调用父窗口的openInGame函数
    if (window.parent && window.parent !== window) {
        console.log('📦 检测到iframe环境');
        try {
            if (typeof window.parent.openInGame === 'function') {
                console.log('✅ 调用父窗口的openInGame函数');
                window.parent.openInGame(pages[page], titles[page]);
                return;
            } else {
                console.warn('⚠️ 父窗口没有openInGame函数，尝试直接跳转');
            }
        } catch (e) {
            console.error('❌ 调用父窗口函数失败:', e);
            console.log('🔄 降级为直接跳转');
        }
    }
    
    // 直接跳转（不在iframe中或父窗口函数不可用）
    console.log('🔄 直接跳转到:', pages[page]);
    try {
        // 使用replace而不是href，避免历史记录堆积
        window.location.replace(pages[page]);
    } catch (e) {
        console.error('❌ 页面跳转失败:', e);
        alert('❌ 页面跳转失败，请刷新重试');
    }
}
```
